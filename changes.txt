+def _match_dofs(vec, dof: int):
+    """Return vec as np.ndarray of length == dof. If vec is longer by 1 (gripper), drop the last."""
+    if vec is None:
+        return None
+    v = np.asarray(vec, dtype=float).reshape(-1)
+    if v.size == dof + 1:
+        v = v[:dof]          # drop gripper
+    elif v.size > dof:
+        v = v[:dof]          # be safe
+    elif v.size < dof:
+        v = np.pad(v, (0, dof - v.size))
+    return v
